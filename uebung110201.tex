\subsection*{Exercise 36}
\begin{algorithmic}[1]
	\Function{LCSubstr}{$S[1..m]$, $T[1..n]$}
    \State $L \gets \mathrm{array}(1..m, 1..n)$
    \State $z \gets 0$
    \State $\mathrm{ret} \gets \emptyset$
    \For{$i \gets 1..m$}
        \For{$j \gets 1..n$}
            \If{$S[i] = T[j]$}
                \If{$i = 1$ or $j = 1$}
                    \State $L[i,j] \gets 1$
                \Else
                    \State $L[i,j] \gets L[i-1,j-1] + 1$
                \EndIf
                \If{$L[i,j] > z$}
                    \State $z \gets L[i,j]$
                    \State $ret \gets \emptyset$
                \EndIf
                \If{$L[i,j] = z$}
                    \State $ret \gets ret \cup {S[i-z+1..i]}$
                \EndIf
        	\EndIf
        \EndFor
    \EndFor
    \Return
    \EndFunction
\end{algorithmic}
